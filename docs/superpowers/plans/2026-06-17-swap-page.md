# Swap Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an Ethereum Mainnet `/swap` page for bidirectional swaps between ETH/USDT/USDC and RWA (plus arbitrary ERC20 via manual address), powered by `@bric-labs/bric-sdk@^0.3.8`.

**Architecture:** Thin `SwapPage` UI + `src/lib/swap/` modules. Quotes and execution go through `BricAggregatorHelper.previewSwapExactInput` / `swapExactInput`. ERC20 approvals use SDK `ERC20Helper`. Token eligibility rules live in pure functions in `swapTokenRules.ts`. Static RWA list in `src/fixtures/SwapTokenWhitelist.ts`.

**Tech Stack:** React 18, TypeScript, ethers v6, Reown AppKit hooks, `@bric-labs/bric-sdk@0.3.8`, sonner toasts.

**Spec:** `docs/superpowers/specs/2026-06-17-swap-page-design.md`

```ts
import { SWAP_PAY_TOKENS, type SwapPayToken } from "@/config/SwapPayTokens";

export type TokenSide = {
  kind: "pay" | "whitelist" | "custom";
  /** pay-token id, whitelist symbol, or checksummed custom address */
  key: string;
  tokenAddress: string;
  symbol: string;
  decimals: number;
};

export const SLIPPAGE_PRESETS = [
  { label: "0.1%", bps: 10, decimal: 0.001 },
  { label: "0.5%", bps: 50, decimal: 0.005 },
  { label: "1%", bps: 100, decimal: 0.01 },
  { label: "3%", bps: 300, decimal: 0.03 },
  { label: "5%", bps: 500, decimal: 0.05 },
] as const;

export const DEFAULT_SLIPPAGE = SLIPPAGE_PRESETS[1]; // 0.5%

export function addressesEqual(a: string, b: string): boolean {
  return a.toLowerCase() === b.toLowerCase();
}

export function isSameTokenSide(a: TokenSide, b: TokenSide): boolean {
  return addressesEqual(a.tokenAddress, b.tokenAddress);
}

export function payTokenSide(token: SwapPayToken): TokenSide {
  return {
    kind: "pay",
    key: token.id,
    tokenAddress: token.tokenAddress,
    symbol: token.symbol,
    decimals: token.decimals,
  };
}

/** Which receive options are valid given current pay side. */
export function getAllowedReceiveSides(pay: TokenSide): TokenSide[] {
  const payIsStableOrEth = pay.kind === "pay";
  const out: TokenSide[] = [];

  if (payIsStableOrEth) {
    // Pay = USDT/USDC → may receive ETH
    if (pay.symbol === "USDT" || pay.symbol === "USDC") {
      out.push(payTokenSide(SWAP_PAY_TOKENS[0])); // ETH
    }
    // Pay = ETH → may receive stables (not ETH)
    if (pay.symbol === "ETH") {
      out.push(payTokenSide(SWAP_PAY_TOKENS[1])); // USDT
      out.push(payTokenSide(SWAP_PAY_TOKENS[2])); // USDC
    }
    // Pay stable → other stable (not self)
    if (pay.symbol === "USDT") {
      out.push(payTokenSide(SWAP_PAY_TOKENS[2]));
    }
    if (pay.symbol === "USDC") {
      out.push(payTokenSide(SWAP_PAY_TOKENS[1]));
    }
    // RWA whitelist entries added by UI layer
    return out.filter((r) => !isSameTokenSide(r, pay));
  }

  // Sell: RWA/custom pay → receive only ETH/USDT/USDC
  return SWAP_PAY_TOKENS.map(payTokenSide).filter(
    (r) => !isSameTokenSide(r, pay)
  );
}

export function isReceiveAllowed(pay: TokenSide, receive: TokenSide): boolean {
  if (isSameTokenSide(pay, receive)) return false;

  if (pay.kind === "pay") {
    // pay-token → receive pay-token from allowed set, whitelist, or custom
    const allowedPayReceives = getAllowedReceiveSides(pay);
    if (receive.kind === "pay") {
      return allowedPayReceives.some((r) => r.key === receive.key);
    }
    // whitelist + custom always OK when buying (unless same address)
    return receive.kind === "whitelist" || receive.kind === "custom";
  }

  // sell: receive must be pay token only
  return receive.kind === "pay";
}

/** After flipping ↕, pick first valid receive if current is invalid. */
export function coerceReceiveSide(
  pay: TokenSide,
  receive: TokenSide,
  whitelistFallback: TokenSide
): TokenSide {
  if (isReceiveAllowed(pay, receive)) return receive;
  const allowed = getAllowedReceiveSides(pay);
  if (allowed.length > 0) return allowed[0];
  if (pay.kind !== "pay") return payTokenSide(SWAP_PAY_TOKENS[0]);
  return whitelistFallback;
}
```

### Task 4: BricSwap service wrapper

**Files:**

- [ ] **Step 1:** Create `src/lib/swap/BricSwap.ts`

```ts
import {
  BricAggregatorHelper,
  ERC20Helper,
  TxStatus,
  getContractAddresses,
  type CallResult,
  type SwapRouterDataOutput,
} from "@bric-labs/bric-sdk";
import { ZeroAddress, type Signer } from "ethers";
import {
  BRIC_MAINNET_CHAIN_ID,
  BRIC_SWAP_AGGREGATOR_ADDRESS,
  initBricSdk,
} from "@/config/BricConfig";

export type SwapQuoteResult = SwapRouterDataOutput & {
  minReceived: bigint;
};

export async function createBricAggregator(
  signer: Signer
): Promise<BricAggregatorHelper> {
  initBricSdk();
  const helper = new BricAggregatorHelper(
    signer,
    BRIC_MAINNET_CHAIN_ID,
    BRIC_SWAP_AGGREGATOR_ADDRESS,
    { waitForConfirmation: true, autoGasBuffer: true }
  );
  return helper.connect(signer, true);
}

export async function fetchSwapQuote(params: {
  signer: Signer;
  tokenIn: string;
  tokenOut: string;
  amountIn: bigint;
  slippageDecimal: number;
  from: string;
}): Promise<SwapQuoteResult> {
  const aggregator = await createBricAggregator(params.signer);
  const quote = await aggregator.previewSwapExactInput(
    params.tokenIn,
    params.amountIn,
    params.tokenOut,
    params.slippageDecimal,
    params.from,
    true
  );
  if (quote.error) {
    throw new Error(quote.error);
  }
  if (!quote.swapData) {
    throw new Error("No swap route found");
  }
  return {
    ...quote,
    minReceived: quote.minReceived ?? 0n,
  };
}

export async function ensureErc20Allowance(params: {
  signer: Signer;
  token: string;
  owner: string;
  amount: bigint;
}): Promise<CallResult | null> {
  if (params.token === ZeroAddress) return null;

  const provider = params.signer.provider;
  if (!provider) throw new Error("Signer has no provider");

  const { BRIC_MULTICALL } = getContractAddresses(BRIC_MAINNET_CHAIN_ID);
  const erc20Helper = new ERC20Helper(provider, BRIC_MULTICALL, true).connect(
    params.signer
  );
  const spender = BRIC_SWAP_AGGREGATOR_ADDRESS;
  const [row] = await erc20Helper.batchBalancesAndAllowances(
    [params.token],
    spender,
    params.owner
  );
  if (row.allowance >= params.amount) return null;

  return erc20Helper.approve(params.token, spender, params.amount);
}

export async function executeSwapExactInput(params: {
  signer: Signer;
  tokenIn: string;
  amountIn: bigint;
  tokenOut: string;
  quote: SwapQuoteResult;
  receiver: string;
}): Promise<CallResult> {
  const aggregator = await createBricAggregator(params.signer);
  aggregator.setOptions({ waitForConfirmation: true, autoGasBuffer: true });

  const minOut = params.quote.minReceived;
  const result = await aggregator.swapExactInput(
    params.tokenIn,
    params.amountIn,
    params.tokenOut,
    minOut,
    params.receiver,
    params.quote.swapData!
  );

  if (result.status === TxStatus.Reverted) {
    throw new Error(result.error?.message ?? "Swap reverted");
  }
  return result;
}
```
